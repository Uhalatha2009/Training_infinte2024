--1--	Write a T-SQL Program to find the factorial of a given number.

create or alter proc factorial
@num int
as
begin
declare @i int = 1
declare @s int = 1
while   @i <=@num
	begin
	set @s=@s*@i
	set @i = @i+1
	end
end
print ' factorial of given number is ' + cast(@s as varchar(10))

factorial 6
